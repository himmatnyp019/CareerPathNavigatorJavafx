package test;

import javafx.application.Application;
import javafx.geometry.Insets;
import javafx.geometry.Pos;
import javafx.scene.Scene;
import javafx.scene.control.Label;
import javafx.scene.image.Image;
import javafx.scene.image.ImageView;
import javafx.scene.layout.GridPane;
import javafx.scene.layout.VBox;
import javafx.scene.text.Font;
import javafx.stage.Stage;

public class MenuSelection extends Application {

    @Override
    public void start(Stage primaryStage) {
        // Create main grid layout
        GridPane grid = new GridPane();
        grid.setAlignment(Pos.CENTER);
        grid.setHgap(20);
        grid.setVgap(20);
        grid.setPadding(new Insets(25, 25, 25, 25));
        grid.setStyle("-fx-background-color: #f5f5f5;");

        // Title
        Label title = new Label("Restaurant Menu");
        title.setFont(Font.font("Arial", 24));
        title.setStyle("-fx-font-weight: bold; -fx-text-fill: #333;");
        
        VBox mainLayout = new VBox(20);
        mainLayout.setAlignment(Pos.CENTER);
        mainLayout.setPadding(new Insets(20));
        mainLayout.getChildren().add(title);

        // Menu items data (name and image path)
        String[][] menuItems = {
            {"Burger", "file:///C:/Users/julfi/Downloads/burger.jpg"},
            {"Pizza", "file:///C:/Users/julfi/Downloads/pizza.jpg"},
            {"Pasta", "file:///C:/Users/julfi/Downloads/pasta.jpg"},
            {"Salad", "file:///C:/Users/julfi/Downloads/salad.jpg"},
            {"Steak", "file:///C:/Users/julfi/Downloads/steak.jpg"},
            {"Ice Cream", "file:///C:/Users/julfi/Downloads/icecream.jpg"},
        };

        // Create menu items
        for (int i = 0; i < menuItems.length; i++) {
            String itemName = menuItems[i][0];
            String imageUrl = menuItems[i][1];
            
            // Create image view
            ImageView imageView = new ImageView();
            try {
                Image image = new Image(imageUrl, 150, 100, true, true);
                imageView.setImage(image);
            } catch (Exception e) {
                // If image fails to load, create a placeholder
                imageView.setStyle("-fx-background-color: #ddd; -fx-min-width: 150; -fx-min-height: 100;");
            }
            
            imageView.setFitWidth(150);
            imageView.setFitHeight(100);
            
            // Create label for item name
            Label nameLabel = new Label(itemName);
            nameLabel.setFont(Font.font("Arial", 16));
            nameLabel.setStyle("-fx-font-weight: bold; -fx-text-fill: #333;");
            nameLabel.setAlignment(Pos.CENTER);
            
            // Create VBox for each menu item
            VBox menuItem = new VBox(10);
            menuItem.setAlignment(Pos.CENTER);
            menuItem.setStyle("-fx-background-color: white; -fx-padding: 15; -fx-border-color: #ddd; -fx-border-radius: 5;");
            menuItem.getChildren().addAll(imageView, nameLabel);
            
            // Add to grid (2 columns, 3 rows)
            grid.add(menuItem, i % 2, i / 2);
        }

        mainLayout.getChildren().add(grid);
        
        // Create scene and show stage
        Scene scene = new Scene(mainLayout, 800, 600);
        primaryStage.setTitle("Restaurant Menu");
        primaryStage.setScene(scene);
        primaryStage.show();
    }

    public static void main(String[] args) {
        launch(args);
    }
}
